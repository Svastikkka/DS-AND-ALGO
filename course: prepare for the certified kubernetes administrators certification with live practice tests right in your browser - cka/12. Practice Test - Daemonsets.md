Deploy a DaemonSet for FluentD Logging.

```bash
kubectl create deploy elasticsearch --image=registry.k8s.io/fluentd-elasticsearch:1.20 -n kube-system -o yaml --dry-run=client > es.yaml
```
Above command will create deployment we need to edit `kind: DaemonSet` and remove `replicas` and `strategy`

```bash
apiVersion: apps/v1
kind: DaemonSet
metadata:
  creationTimestamp: null
  labels:
    app: elasticsearch
  name: elasticsearch
  namespace: kube-system
spec:
  selector:
    matchLabels:
      app: elasticsearch
  template:
    metadata:
      creationTimestamp: null
      labels:
        app: elasticsearch
    spec:
      containers:
      - image: registry.k8s.io/fluentd-elasticsearch:1.20
        name: fluentd-elasticsearch
        resources: {}
status: {}
```
